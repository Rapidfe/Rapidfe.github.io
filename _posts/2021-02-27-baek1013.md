#### **오답 1**

다음 state가 확정적이지 않을 때는 몇 개의 state를 추가로 만들어서 다음 수를 보고 판단해야한다.



# **코드**

---
{% raw %}
```c++
#include <iostream>
#include <vector>

using namespace std;

int nextn[9][2] = {{7,1}, {2,-1}, {3,-1},
                   {3,4}, {7,5}, {6,5},
                   {3,8}, {-1,8}, {7,1}};
int cases,state;
string str;

void check(){
    int num;
    state = 0;
    for(int i=0; i<str.length(); i++){
        num = str[i]-'0';
        state = nextn[state][num];
        if( state==-1 ){
            cout<<"NO\n";
            return;
        }
    }
    if( state==4 || state==5 || state==8 ) cout<<"YES\n";
    else cout<<"NO\n";
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cin>>cases;
    for(int i=0; i<cases; i++){
        cin>>str;
        check();
    }
    return 0;
}
```
